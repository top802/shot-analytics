CREATE TABLE `shot_analytics`.`analytics` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `date_picker` VARCHAR(25) NOT NULL,
  `strafing` INT NOT NULL,
  `numbersCannonades` INT NOT NULL,
  `startStrafing` TIME NOT NULL,
  `endStrafing` TIME NOT NULL,
  `position` VARCHAR(45) NOT NULL,
  `weapon_type` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC)
);

INSERT INTO `shot_analytics`.`analytics`
 (`id`, `date_picker`, `strafing`, `numbersCannonades`,
  `startStrafing`, `endStrafing`, `position`, `weapon_type`)
   VALUES ('1', '2023-06-14', '1', '2', '10:00', '10:30', 'Don', 'Arta');

// get count strafing and count cannonades per day on a position
SELECT
    date_picker,
    position,
    CONCAT(
        TIME_FORMAT(MIN(startStrafing), '%H:%i'),
        ' - ',
        TIME_FORMAT(DATE_ADD(MIN(startStrafing), INTERVAL 1 HOUR), '%H:%i')
    ) AS strafingTime,
    SUM(strafing) AS strafing,
    SUM(numbersCannonades) AS numbersCannonades
FROM
    `shot_analytics`.`analytics`
WHERE
    date_picker = '2023-06-18'
    AND position = 'Don'
GROUP BY
    date_picker,
    position,
    HOUR(startStrafing)
ORDER BY
    date_picker,
    position,
    HOUR(startStrafing)
